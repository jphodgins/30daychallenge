function getCurrentDay() {

const now = new Date();

now.setHours(0,0,0,0);

const start = new Date(START_DATE);

start.setHours(0,0,0,0);

const diff = Math.floor((now - start) / 86400000) + 1;

return Math.max(1, Math.min(diff, TOTAL_DAYS));

}


function getWeekNumber(dayNum) {

function getDayDate(dayNum) {

return d;

}

function getDailyCount(user, day) {

const data = getUserDayData(user, day);

return DAILY_ACTIVITIES.filter(a => !!data[a.id]).length;

}


function getDailyTotal() {

return DAILY_ACTIVITIES.length;

}


function isWeeklyDone(user, weekNum, activityId) {

const startDay = (weekNum - 1) * 7 + 1;

const endDay = Math.min(weekNum * 7, TOTAL_DAYS);

for (let d = startDay; d <= endDay; d++) {

if (isChecked(user, d, activityId)) return true;

}

return false;

}


function getStreak(user, activityId) {

let streak = 0;

const today = getCurrentDay();

for (let d = today; d >= 1; d--) {

if (ACTIVITIES.find(a => a.id === activityId).freq === 'weekly') {

const week = getWeekNumber(d);

if (isWeeklyDone(user, week, activityId)) {

// Count all days of this week as part of streak and skip to prev week

streak++;

d = (week - 1) * 7 + 1; // will decrement to last day of prev week

} else {

break;

}

} else {

if (isChecked(user, d, activityId)) {

streak++;

} else if (d < today) {

break; // allow today to be unchecked

} else {

break;

}

}

}

return streak;

}


function getOverallStreak(user) {

let streak = 0;

const today = getCurrentDay();

for (let d = today; d >= 1; d--) {

const count = getDailyCount(user, d);

if (count === getDailyTotal()) {

streak++;

} else if (d < today) {

break;

} else {

break;

}

}

return streak;

}


function getTotalPoints(user) {

let points = 0;

for (let d = 1; d <= TOTAL_DAYS; d++) {

const data = getUserDayData(user, d);

ACTIVITIES.forEach(a => {

if (data[a.id]) points++;

});

}

return points;

}


function getCompletedDays(user) {

let count = 0;

for (let d = 1; d <= TOTAL_DAYS; d++) {

if (getDailyCount(user, d) === getDailyTotal()) count++;

}

return count;

}

<div class="activity-name">${activity.icon} ${activity.name}</div>

<div class="activity-freq">Daily</div>

</div>

${streak > 1 ? `<div class="activity-streak">🔥 ${streak} day streak</div>` : ''}

`;

grid.appendChild(row);

});


// Weekly activities

WEEKLY_ACTIVITIES.forEach(activity => {

const weekDone = isWeeklyDone(currentUser, weekNum, activity.id);

const dayChecked = isChecked(currentUser, selectedDay, activity.id);

const streak = getStreak(currentUser, activity.id);

const row = document.createElement('div');

row.className = 'activity-row' + (dayChecked ? ' checked' : '');

row.onclick = () => toggleActivity(activity.id);

row.innerHTML = `

<div class="checkbox"></div>

<div class="activity-info">

<div class="activity-name">${activity.icon} ${activity.name}</div>

<div class="activity-freq">Once per week ${weekDone && !dayChecked ? ' · ✅ Done this week' : ' · Week ' + weekNum}</div>

</div>

${streak > 1 ? `<div class="activity-streak">🔥 ${streak} week streak</div>` : ''}

`;

grid.appendChild(row);

});


// Encouragement

const dailyDone = getDailyCount(currentUser, selectedDay);

if (dailyDone === getDailyTotal()) {

showToast("All daily activities complete! " + ENCOURAGEMENTS[selectedDay % ENCOURAGEMENTS.length]);

}

}

const streak = getOverallStreak(user);


const card = document.createElement('div');

card.className = 'person-card' + (user === currentUser ? ' selected' : '');

card.innerHTML = `

<div class="person-progress-bar"><div class="person-progress-fill" style="width:${pct}%"></div></div>

<div class="person-stats">${done}/${total} today · ${pct}%</div>

${streak > 0 ? `<div class="person-streak-badge">🔥 ${streak} day streak</div>` : ''}

`;

cardsContainer.appendChild(card);

});


// Detail table

document.getElementById('groupDetailTitle').textContent = `Day ${today} — ${formatDate(getDayDate(today))}`;

tbody.innerHTML = '';


ACTIVITIES.forEach(activity => {

const tr = document.createElement('tr');

const isWeekly = activity.freq === 'weekly';

let cells = `<td>${activity.icon} ${activity.name}${isWeekly ? ' <span style="color:var(--text-secondary);font-size:0.75rem">(weekly)</span>' : ''}</td>`;


USERS.forEach(user => {

let done;

if (isWeekly) {

done = isWeeklyDone(user, getWeekNumber(today), activity.id);

} else {

done = isChecked(user, today, activity.id);

}

cells += `<td><span class="status-dot ${done ? 'done' : ''}"></span></td>`;

});


tr.innerHTML = cells;

tbody.appendChild(tr);

});

}

let toastTimeout;

function showToast(msg) {

const t = document.getElementById('toast');

t.classList.add('show');

clearTimeout(toastTimeout);

toastTimeout = setTimeout(() => t.classList.remove('show'), 3000);

}

const milestone = MILESTONES.find(m => m.threshold === days);

if (milestone) {

setTimeout(() => showToast(`${milestone.icon} ${milestone.label}!`), 500);

}

}
